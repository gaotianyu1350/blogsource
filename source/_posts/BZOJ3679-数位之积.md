## 代码
```c++
#include <cstdio>
#include <cstring>
#include <cmath>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <queue>
#include <map>
using namespace std;

typedef long long ll;
const int MaxS = 1e5;
const int MaxLen = 20;

inline void update(ll &a, ll b) {
    a += b;
}

namespace dp {
    ll num[MaxS];
    ll sum[MaxS], f[2][2][MaxS];
    int dy[MaxS][10];
    int tot;

    inline void init(ll n) {
        for (ll i = 1; i <= n; i *= 2)
            for (ll j = i; j <= n; j *= 3)
                for (ll k = j; k <= n; k *= 5)
                    for (ll p = k; p <= n; p *= 7)
                        num[++tot] = p;
        sort(num + 1, num + 1 + tot);
        for (int i = 1; i <= tot; i++) {
            ll s = num[i];
            for (int j = 1; j <= 9; j++)
                if (s % j == 0) {
                    dy[i][j] = lower_bound(num + 1, num + 1 + tot, s / j) - num;
                    if (dy[i][j] == tot + 1 || num[dy[i][j]] != s / j)
                        dy[i][j] = 0;
                }
        }
    }

    inline ll solve(ll n) {
        if (!n) return 0;
        int now = 1, pre = 0;
        memset(f, 0, sizeof(f));
        memset(sum, 0, sizeof(sum));
        f[pre][0][1] = f[pre][1][1] = 1;
        ll tmp = n;
    
        while (tmp) {
            int curBit = tmp % 10;
            tmp /= 10;
            for (int t = 1; t <= tot; t++) {
                f[now][0][t] = f[now][1][t] = 0;
                // Limit
                for (int i = 1; i < curBit; i++)
                    update(f[now][0][t], f[pre][1][dy[t][i]]);
                update(f[now][0][t], f[pre][0][dy[t][curBit]]);
                // Not Limit
                for (int i = 1; i < 10; i++)
                    update(f[now][1][t], f[pre][1][dy[t][i]]);
                
                if (tmp)
                    update(sum[t], f[now][1][t]);
                else
                    update(sum[t], f[now][0][t]);
            }
            swap(now, pre);
        }

        ll ans = 0;
        for (int i = 1; i <= tot; i++)
            ans += sum[i];
        return ans;
    }

};

int main() {
    ll n, l, r;
    cin >> n >> l >> r;
    dp::init(n);
    ll ans = dp::solve(r - 1) - dp::solve(l - 1);
    cout << ans << endl;
}
```
